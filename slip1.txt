const http =require('http');
const str ="Full Stack";
http.createServer((req,res)=>{
    const newsttr=str.split('').reverse().join('');
    res.write(`${newsttr} \n`);
    const n1=str.toUpperCase();
    res.write(`${n1} \n`);
    const n2=str.toLowerCase();
    res.write(`${n2} \n`);
    res.end();
}).listen(3099);
